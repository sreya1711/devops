1. Update system packages
sudo apt update
sudo apt upgrade -y

2. Install Docker
sudo apt install docker.io -y

3. Check Docker version
docker --version

4. Start Docker service
sudo systemctl start docker

5. Enable Docker at boot
sudo systemctl enable docker

6. Check Docker status
sudo systemctl status docker

7. Run Nginx container
sudo docker run -d -p 8080:80 nginx

8. Check running containers
sudo docker ps

9. Create a custom index.html file
mkdir ~/mynginx
cd ~/mynginx
vim index.html

10. Add HTML content inside index.html

<html>
<head>
<title>Day 3 Docker Nginx</title>
</head>
<body>
<h1>Docker Nginx Setup Successful</h1>
</body>
</html>

Save and exit (:wq)

11. Run Nginx container with custom index.html
sudo docker run -d -p 8081:80 -v ~/mynginx:/usr/share/nginx/html nginx

12. Open in browser
http://localhost:8081

13. Stop container
sudo docker stop <container_id>

14. Remove container
sudo docker rm <container_id>
